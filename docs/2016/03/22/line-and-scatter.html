import co.theasi.plotly import util.Random // Generate uniformly distributed x val xs = (0 until 100) // Generate random y val ys0 = (0 until 100).map { i => Random.nextDouble } val ys1 = ys0.map { _ + 5.0 } val ys2 = ys0.map { _ - 5.0 } val p = Plot() .withScatter(xs, ys0, ScatterOptions().mode(ScatterMode.Marker).name("marker")) .withScatter(xs, ys1, ScatterOptions() .mode(ScatterMode.Marker, ScatterMode.Line) .name("line+marker")) .withScatter(xs, ys2, ScatterOptions().mode(ScatterMode.Line).name("line")) draw(p, "scatter-mode", writer.FileOptions(overwrite=true))
