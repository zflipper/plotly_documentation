library(plotly) df <- data.frame(x = sample(LETTERS[1:10], size = 1000, replace = T), y = sample(1:100, size = 1000, replace = T), group = sample(letters[1:5], size = 1000, replace = T)) p <- ggplot(df, aes(x, y, color = group)) + geom_point() + geom_jitter(width = 1, height = 1) + ggtitle("geom_jitter: scatterplots") ggplotly(p)
